        const m = 12;
        const table = document.querySelector("table");

        function randint(a, b) {
            return Math.floor(Math.random() * (b - a + 1)) + a;
        }

        function start() {
            fox = { x: n - 1, y: m - 1 }; 
            rabbit = { x: randint(0, n - 1), y: 0 }; // ny√∫l a bal oldalon
            showBoard();
            poziciok();
            button.disabled = true;  
            level(levelCount);
        }

        function showBoard() {
            table.innerHTML = "";
            for (let i = 0; i < n; i++) {
                const tr = document.createElement("tr");
                for (let j = 0; j < m; j++) {
                    const td = document.createElement("td");
                    td.innerText = "";
                    tr.appendChild(td);
                }
                table.appendChild(tr);
            }
        }

        function poziciok() {
            table.rows[fox.x].cells[fox.y].innerText = "ü¶ä";
            table.rows[rabbit.x].cells[rabbit.y].innerText = "üê∞";
        }

        function handleMove(event) {
            const key = event.key;
            if (key === "ArrowUp" && fox.x > 0) {
                fox.x--;
            } else if (key === "ArrowDown" && fox.x < n - 1) {
                fox.x++;
            } else if (key === "ArrowLeft" && fox.y > 0) {
                fox.y--;
            } else if (key === "ArrowRight" && fox.y < m - 1) {
                fox.y++;
            }
            showBoard();
            poziciok();
        }

            rabbit.x = randint(0, n - 1);

            poziciok();
